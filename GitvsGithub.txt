# Git Commands Cheat Sheet

## Basic Commands

# Check Git version

git --version

# Check repository status

git status

# Set global email

git config --global user.email "[your-email@example.com](mailto:your-email@example.com)"

# Set global username

git config --global user.name "your name"

# Show all Git configurations

git config --list

# Initialize repository

git init

## Add & Commit

# Add specific files

git add <file1> <file2>

# Add all files

git add .

# Commit changes

git commit -m "write message"

# Commit tracked files directly

git commit -am "message"

## Logs & History

# Show commit history

git log

# One-line history

git log --oneline

# Show branch graph

git log --oneline --graph --all

## Branching

# Show branches

git branch

# Create branch

git branch <branch-name>

# Switch branch

git switch <branch-name>

# Create & switch branch together

git switch -c <branch-name>

# Old method to switch branch

git checkout <branch-name>

# Go back to main branch

git checkout main

# OR

git switch main

## Merge & Rebase

# Merge branch

git merge <branch-name>

# Rebase branch

git rebase <branch-name>

# Cherry-pick commit

git cherry-pick <commit-id>

## Rename & Delete Branch

# Rename branch

git branch -m <old-branch-name> <new-branch-name>

# Rename current branch

git branch -m <new-name>

# Safe delete branch

git branch -d <branch-name>

# Force delete branch

git branch -D <branch-name>

## Remote Repository Commands

# Clone repository

git clone <repository-url>

# Connect remote repository

git remote add origin <repository-url>

# Show remote repositories

git remote -v

# Push code to GitHub

git push origin main

# Push new branch

git push -u origin <branch-name>

# Pull latest code

git pull origin main

# Fetch latest changes

git fetch

## File Management

# See differences

git diff

# Remove file from staging

git restore --staged <file-name>

# Restore deleted/modified file

git restore <file-name>

# Remove tracked file

git rm <file-name>

# Rename file

git mv <old-name> <new-name>

## Commit Management

# View commit details

git show <commit-id>

# Undo last commit (keep changes)

git reset --soft HEAD~1

# Undo last commit completely

git reset --hard HEAD~1

## Stash Commands

# Save temporary work

git stash

# Restore stashed work

git stash pop

# Show stash list

git stash list

## Tags

# Create tag

git tag v1.0

# Show tags

git tag

# Push tags

git push origin --tags
